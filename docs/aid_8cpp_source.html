420 void aid::initialize_from_string(const string& init_aid)
421 {
422 vector<string> values;
423 boost::split(values,init_aid,boost::is_any_of(":"));
424 string aid_type;
425 if (values[0].front() == '(' and isalpha(values[0][1]))
426 {
427 string new_universe;
428 if (isalpha(values[0][2]))
429 {
430 astronomical_type = create_astronomical_type(values[0].substr(1,2));
431 if (values[0][4] != ' ')
432 {
433 new_universe = values[0].substr(4);
434 }
435 else
436 {
437 new_universe = values[0].substr(5);
438 }
439 }
440 else
441 {
442 astronomical_type = create_astronomical_type(values[0].substr(1,1));
443 if (values[0][3] != ' ')
444 {
445 new_universe = values[0].substr(3);
446 }
447 else
448 {
449 new_universe = values[0].substr(4);
450 }
451 }
452 if (values.size() == 4 and astronomical_type != aid::UNIVERSE and astronomical_type != aid::GALAXY and astronomical_type != aid::SOLAR_SYSTEM)
453 {
454 universe = new_universe;
455 galaxy = values[1];
456 solar_system = values[2];
457 astronomical_body = values[3];
458 }
459 else if (values.size() == 3 and astronomical_type == aid::SOLAR_SYSTEM)
460 {
461 universe = new_universe;
462 galaxy = values[1];
463 solar_system = values[2];
464 }
465 else if (values.size() == 2 and astronomical_type == aid::GALAXY)
466 {
467 universe = new_universe;
468 galaxy = values[1];
469 }
470 else if (values.size() == 1 and astronomical_type == aid::UNIVERSE)
471 {
472 universe = new_universe;
473 }
474 if (universe == "")
475 {
476 astronomical_type = aid::NONE;
477 }
478 }
